168        /**
169         * Converts an image to a "disabled" version.
170         *
171         * @param img The image to "disable".
172         */
173        private static void makeDisabledImage(GreenfootImage img) {
174            if (img == null) return;
175            final float RED_LUMINANCE = 0.229f;
176            final float GREEN_LUMINANCE = 0.587f;
177            final float BLUE_LUMINANCE = 0.114f;
178            final int RGB_MIN = 148;
179            final int RGBA_MAX = 255;
180            for (int x = 0; x < img.getWidth(); x++) {
181                for (int y = 0; y < img.getHeight(); y++) {
182                    Color c = img.getColorAt(x, y);
183                    if (c.getAlpha() >= RGBA_MAX) {
184                        float red = c.getRed() * RED_LUMINANCE;
185                        float green = c.getGreen() * GREEN_LUMINANCE;
186                        float blue = c.getBlue() * BLUE_LUMINANCE;
187                        int luminance = (int) (red + green + blue);
188                        if (luminance < RGB_MIN) luminance = RGB_MIN;
189                        Color color = new Color(luminance, luminance, luminance);
190                        img.setColorAt(x, y, color);
191                    }
192                }
193            }
194        }

417        /**
418         * Requests that this component get the input focus.
419         *
420         * To receive focus, this component must be focusable and enabled.
421         * Developers must never assume that this component is the focus owner
422         * and must check the return value to be certain. If this method returns
423         * true then focus was received. If <code>false</code> is returned,
424         * the request failed.
425         *
426         * @return <code>true</code> if the focus succeeds and <code>false</code>
427         *         if the focus fails.
428         */
429        public boolean requestFocus() {
430            if (isFocusable() && isEnabled()) {
431                if (focusOwner != null) {
432                    GUIComponent oldFocusOwner = focusOwner;
433                    focusOwner = null;
434                    oldFocusOwner.repaint(); // to remove focus decoration
435                }
436                // Clear KB buffer disabled due to null pointer on start up
437                // java.lang.NullPointerException
438                //  at greenfoot.Greenfoot.getKey(Greenfoot.java:86)
439                // Evidently, Greenfoot is not always available when World is
440                // Greenfoot.getKey(); // clear keyboard buffer
441                focusOwner = this;
442                focusOwner.repaint(); // add focus decoration
443                return true;
444            }
445            return false;
446        }

453         * @since 1.4
454         */
455        public boolean isFocusOwner() {
456            return focusOwner == this;

557        /**
558         * Wraps a single string into an array of strings for some maximum
559         * number of characters. Also will split lines on newlines or the
560         * characters "\n" allowing simple manually placed newlines.
561         *
562         * @param str The text to word wrap.
563         * @param max The maximum number of characters per line.
564         * @return The word-wrapped lines of text.
565         *
566         * @author http://joust.kano.net/weblog/archives/000060.html
567         * @author Ed Parrish (changed regex, added max, added newline split,
568         * removed extra blank line at end)
569         */
570        public static String[] wordWrap(String str, int max) {
571            if (max <= 0) return new String[0];
572            Pattern wrapRE =
573                Pattern.compile(".{0," + (max - 1) + "}(?:\\S(?:-| |\n|$)|\n|$)");
574            List<String> list = new LinkedList<String>();
575            Matcher m = wrapRE.matcher(str);
576            while (m.find()) list.add(m.group());
577            if (list.get(list.size() - 1).equals("")) list.remove(list.size() - 1);
578            return (String[]) list.toArray(new String[list.size()]);
579        }

581        /**
582         * Computes the dimension needed to display text in the specified
583         * <code>Graphics</code> context.
584         *
585         * @param str The text to display on this component.
586         * @param g The <code>Graphics</code> context in which to paint.
587         * @return the dimension (width and height) needed.
588         */
589        public Dimension getTextDimension(String str, Graphics g) {
590            // Calculate image size based on text width & height
591            if (str == null) str = "";
592            FontMetrics fm = g.getFontMetrics();
593            // Allow multiple lines separated by newline characters ('\n')
594            String[] lines = str.split("\n");
595            int width = 1;
596            int height = 0;
597            Rectangle2D[] bounds = new Rectangle2D[lines.length];
598            for (int i = 0; i < lines.length; i++) {
599                bounds[i] = g.getFontMetrics().getStringBounds(lines[i], g);
600                width = Math.max(width, (int) Math.ceil(bounds[i].getWidth()));
601                height += Math.ceil(bounds[i].getHeight());
602            }
603            Insets insets = getInsets();
604            if (insets != null) {
605                width += insets.left + insets.right;
606                height += insets.top + insets.bottom;
607            }
608            if (height <= 0) height = 1;
609            return new Dimension(width, height);
610        }

612        /**
613         * Splits the text into multiple lines in the specified
614         * <code>Graphics</code> context.
615         *
616         * If the component size is fixed, then the text is word wrapped to fit.
617         * Otherwise, text is split on '\n' characters.
618         *
619         * @param str The text to display on this component.
620         * @param g The <code>Graphics</code> context in which to paint.
621         * @return the split lines of text.
622         */
623        public String[] splitLines(String str, Graphics g) {
624            FontMetrics fm = g.getFontMetrics();
625            if (str == null) str = "";
626            String[] lines = null;
627            if (isFixedSize()) {
628                int pixelsWide = fm.stringWidth(str);
629                Insets insets = getInsets();
630                if (insets != null) {
631                    pixelsWide += (insets.left + insets.right);
632                }
633                int maxChars = Math.round(getWidth()
634                    / ((float) pixelsWide / str.length()));
635                lines = wordWrap(str, maxChars);
636            } else {
637                lines = str.split("\n");
638            }
639            return lines;
640        }

708    /**
709     * A class which implements a simple line border in the color black.
710     *
711     * See the <code>GUIWorld</code> class for examples of how to use this
712     * class.
713     *
714     * @author Ed Parrish
715     * @version 1.0  01/05/2011
716     */
717    class LineBorder implements Border {
718        // Padding to leave small horizontal gap between box edge and cursor
719        private static final int X_PAD = 3;
720        private Color color;
721        private int thickness;
722    
723        /**
724         * Creates a simple line border in the color black and a thickness of one.
725         */
726        public LineBorder() {
727            this(Color.BLACK, 1);
728        }
729    
730        /**
731         * Creates a simple line border with the specified color and thickness.
732         *
733         * @param lineColor The color of the border.
734         * @param lineThickness The thickness of the border in pixels.
735         */
736        public LineBorder(Color lineColor, int lineThickness) {
737            color = lineColor;
738            thickness = lineThickness;
739        }
740    
741        /**
742         * Paints the border for the specified component.
743         *
744         * @param c The component for which this border is being painted.
745         * @param g The graphics context in which to paint.
746         */
747        public void paintBorder(GUIComponent c, Graphics g) {
748            Color oldColor = g.getColor();
749            g.setColor(color);
750            for (int i = 0; i < thickness; i++)  {
751                g.drawRect(i, i, c.getWidth() - i - i - 1,
752                    c.getHeight() - i - i - 1);
753            }
754            g.setColor(oldColor);
755        }
756    
757        /**
758         * Returns a new <code>Insets</code> object where the <code>top</code>,
