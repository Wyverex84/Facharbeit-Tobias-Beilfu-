174        // Test the isPressed() method of Button
175        public void act() {
176            defaultLabel.setText("isPressed: "+defaultBtn.isPressed());
177    
178            // Put red border around text field when selected
179            Border b = colorTF.getBorder();
180            if (colorTF.isFocusOwner() && b != redBorder) {
181                colorTF.setBorder(redBorder);
182            } else if (!colorTF.isFocusOwner() && b != blueBorder) {
183                colorTF.setBorder(blueBorder);
184            }
185        }
186    
187        // Test the actionPerformed() callback
188        public void actionPerformed(GUIComponent c) {
189            if (c.getID() == 1002) {
190                biggerLabel.setText("action on\nbutton\n# " + c.getID());
191            }
192            if (c != onOffBtn) {
193                c.setText("Action \nPerformed");
194            } else {
195                enabled = !enabled;
196                defaultLabel.setEnabled(enabled);
197                biggerLabel.setEnabled(enabled);
198                colorfulLabel.setEnabled(enabled);
199                fixedLabel.setEnabled(enabled);
200                multilineLabel.setEnabled(enabled);
201                defaultBtn.setEnabled(enabled);
202                biggerBtn.setEnabled(enabled);
203                colorfulBtn.setEnabled(enabled);
204                fixedSizeBtn.setEnabled(enabled);
205                multiLineBtn.setEnabled(enabled);
206                rectBtn.setEnabled(enabled);
207                roundBtn.setEnabled(enabled);
208                ovalBtn.setEnabled(enabled);
209                squareBtn.setEnabled(enabled);
210                multiBtn.setEnabled(enabled);
211                defaultTF.setEnabled(enabled);
212                biggerTF.setEnabled(enabled);
213                colorTF.setEnabled(enabled);
214                fixedSizeTF.setEnabled(enabled);
215                longTF.setEnabled(enabled);
216                if (enabled) {
217                    c.setText("Disable\nComponents");
218                } else {
219                    c.setText("Enable\nComponents");
220                }
221            }
