007    import java.awt.Insets;

014    /**
015     * Class <code>TextField</code> is a Greenfoot component that allows the user
016     * to enter a single line of text.
017     *
018     * When users are finished entering text, they can signal they are finished by
019     * pressing the Enter key. Pressing the Enter key will fire the ActionListener
020     * interface.
021     *
022     * The <code>ActionListener</code> interface is like the
023     * <code>ActionListener</code> interface in Java Swing. Scenarios using the
024     * <code>ActionListener</code> callback must include the
025     * <code>ActionListener</code> interface, and the class listening for button
026     * clicks must implement that interface. In addition, the implementing class
027     * must add itself to the callback list by calling the
028     * <code>addActionListener()</code> method. Then when the Enter key is pressed,
029     * the <code>actionPerformed()</code> method in the listening class will be
030     * called.
031     *
032     * The developer can read the text the user entered by calling the
033     * <code>getText()</code> method.
034     *
035     * @author Ed Parrish
036     * @version 1.0  12/28/2010
037     * @see http://java.sun.com/developer/onlineTraining/Media/2DText/index.html
038     */
039    public class TextField extends GUIComponent {
040        private static final Color CARET_COLOR = Color.RED;
041        private static final Color HIGHLIGHT_COLOR = Color.PINK;
042        private static final long KEYSTOKE_DELAY = 110;
043        private static final long BLINK_DELAY = 400;
044        // Padding to leave small horizontal gap between box edge and cursor
045        private static final int X_PAD = 1;
046        // Input text buffer
047        private StringBuilder input;
048        // Width of the input field
049        private int numCols;
050        // TextLayout is used for drawing and measuring. Since it is expensive to
051        // create, cache it and invalidate it when text is modified.
052        private TextLayout layout = null;
053        private boolean validTextLayout = false;
054        // Timer variable used to prevent keystrokes from going too fast
055        private long lastKeystrokeTime;
056        // The main insertion index.
057        private int index1 = 0;
058        // The second insertion index used for mouse selection
059        private int index2;
060        // The x-coordinate of the origin relative to the left side of the component
061        private float originX;
062        // The y-coordinate of the origin relative to the top of the component
063        private float originY;
064        // Component with the cursor showing
065        private GreenfootImage cursorOn;
066        // Component without the cursor showing
067        private GreenfootImage cursorOff;
068        // Timer variable to control blinking cursor
069        private long lastBlinkTime;
070        // Variable to control blink state
071        private boolean blinkOn;
072

117         * @param columns Number of columns to calculate the width.

121            this (text, columns, font, DEFAULT_FG, Color.WHITE);

124        /**
125         * Constructs a text field of the specified width in columns with the
126         * initial text displayed in the specified font and text color on the
127         * specified background color.
128         *
129         * @param text The initial text to display on the text field.
130         * @param columns Number of columns to calculate the width.
131         * @param font The font used to write on this component.
132         * @param fg The desired foreground color.
133         * @param bg The desired background color.
134         */
135        public TextField(String text, int columns, Font font, Color fg, Color bg) {
136            super(text, font, fg, bg, null);
137            numCols = columns;
138            input = new StringBuilder(text);
139            super.repaint();
140            originX = getInsets().left + X_PAD;
141        }
142    
143        /**
144         * Handle keystrokes and mouse actions for this component.
145         */
146        public void act() {
147            super.act(); // check for focus request
148            if (!isEnabled()) return;
149    
150            // Move caret using mouse pointer
151            if (Greenfoot.mousePressed(this)) {
152                MouseInfo e = Greenfoot.getMouseInfo();
153                if (e != null && layout != null) {
154                    index1 = getHitPosition(e.getX(), e.getY());
155                    index2 = index1;
156                }
157                // Repaint the component so the new caret will be displayed.
158                super.repaint();
159            }
160    
161            // Highlight text with mouse pointer
162            if (Greenfoot.mouseDragged(this)) {
163                MouseInfo e = Greenfoot.getMouseInfo();
164                if (e != null && layout != null) {
165                    index2 = getHitPosition(e.getX(), e.getY());
166                }
167                // Repaint the component so the highlight will be displayed.
168                super.repaint();
169            }
170    
171            // Ignore keystrokes if not the focus owner
172            if (!isFocusOwner()) return;
173    
174            // Blink the cursor
175            if (System.currentTimeMillis() - lastBlinkTime > BLINK_DELAY) {
176                lastBlinkTime = System.currentTimeMillis();
177                blinkOn = !blinkOn;
178                if (blinkOn) {
179                    setImage(cursorOn);
180                } else {
181                    setImage(cursorOff);
182                }
183            }
184    
185            // Slow down keystrokes on fast systems
186            if (System.currentTimeMillis() - lastKeystrokeTime < KEYSTOKE_DELAY) {
187                return;
188            }
189            lastKeystrokeTime = System.currentTimeMillis();
190    
191            // Process keystrokes
192            TextHitInfo newPosition = null;
193            String key = Greenfoot.getKey();
194            if (key != null && key.length() == 1) {
195                // Single character keystrokes like: 'a'...'z', '0'...'9'
196                deleteHighlighedChars();
197                input.insert(index1, key);
198                index1++;
199                index2 = index1;
200                repaint();
201            } else if (Greenfoot.isKeyDown("space")) {
202                deleteHighlighedChars();
203                input.insert(index1, " ");
204                index1++;
205                index2 = index1;
206                repaint();
207            } else if (Greenfoot.isKeyDown("backspace")) {
208                if (index1 != index2) {
209                    deleteHighlighedChars();
210                    repaint();
211                } else if (input.length() > 0 && index1 > 0) {
212                    --index1;
213                    input.deleteCharAt(index1);
214                    index2 = index1;
215                    repaint();
216                }
217            } else if (Greenfoot.isKeyDown("enter")) {
218                fireActionEvent();
219            } else if (Greenfoot.isKeyDown("right") && layout != null) {
220                newPosition = layout.getNextRightHit(index2);
221            } else if (Greenfoot.isKeyDown("left") && layout != null) {
222                newPosition = layout.getNextLeftHit(index1);
223            }
224            if (newPosition != null) {
225                index1 = newPosition.getInsertionIndex();
226                index2 = index1;
227                super.repaint();
228            }
229        }

231        /**
232         * Set the text to be displayed.
233         *
234         * @param newText The new text to be displayed on the button.
235         */
236        public void setText(String newText) {
237            if (newText == null) {
238                input.replace(0, input.length(), "");
239            } else {
240                input.replace(0, input.length(), newText);
241            }
242            index1 = 0;
243            index2 = index1;
244            repaint();
245        }

247        /**
248         * Returns the text for this component.
249         *
250         * @return The text displayed on this component.
251         */
252        public String getText() {
253            return input.toString();
254        }

256        /**
257         * Sets the number of columns in this <code>TextField</code>.
258         *
259         * @param columns the number of columns >= 0
260         * @exception IllegalArgumentException if <code>columns</code>
261         *      is less than 0
262         */
263        public void setColumns(int columns) {
264            int oldVal = numCols;
265            if (columns < 0) {
266                throw new IllegalArgumentException("columns less than zero.");
267            }
268            if (numCols != oldVal) {
269                numCols = columns;
270                repaint();
271            }
272        }
273    
274        /**
275         * Returns the number of columns in this <code>TextField</code>.
276         *
277         * @return the number of columns >= 0.
278         */
279        public int getColumns() {
280            return numCols;
281        }

283        /**
284         * Paints the component image, including the background, border and text.
285         *
286         * This method recreates the text layout. To repaint without recreating
287         * the text layout, call <code>super.repaint()</code>.
288         */
289        public void repaint() {
290            validTextLayout = false;
291            super.repaint();
292        }

294        /**
295         * Paints the text field's background image.
296         *
297         * @param g The <code>Graphics</code> context in which to paint.
298         */
299        public void paintComponent(Graphics g) {
300            GreenfootImage img = null;
301            if (isFixedSize()) {
302                img = new GreenfootImage(getWidth(), getHeight());
303            } else {
304                // Calculate image size based on text height & numCols
305                FontMetrics fm = g.getFontMetrics();
306                String text = getText();
307                if (text == null) text = "";
308                int width = fm.charWidth('m') * numCols;
309                if (width <= 0) width = 1;
310                int height = fm.getHeight();
311                Insets insets = getInsets();
312                if (insets != null) {
313                    width += insets.left + insets.right;
314                    height += insets.top + insets.bottom;
315                }
316                img = new GreenfootImage(width, height);
317            }
318            if (getBackground().getTransparency() > 0) {
319                img.setColor(getBackground());
320                img.fill();
321            }
322            // Paint box around the text field if no border supercedes it
323            if (getBorder() == null) {
324                img.setColor(Color.BLACK);
325                img.drawRect(0, 0, img.getWidth() - 1, img.getHeight() - 1);
326                // Draw thicker line if this component has the focus
327                if (isFocusOwner()) {
328                    img.drawRect(1, 1, img.getWidth() - 3, img.getHeight() - 3);
329                }
330            }
331            setImage(img);
332        }

334        /**
335         * Paints the text onto the labels background image.
336         *
337         * @param g The <code>Graphics</code> context in which to paint.
338         */
339        public void paintText(Graphics g) {
340            Graphics2D g2 = (Graphics2D) g;
341            // Draw the text that the user has entered
342            TextLayout tl = getTextLayout(g2);
343            if (tl != null) {
344                scrollToVisible();
345                AffineTransform at
346                    = AffineTransform.getTranslateInstance(originX, originY);
347    
348                if (isFocusOwner()) {
349                    // Draw the highlight
350                    Shape hilight = tl.getLogicalHighlightShape(index1, index2);
351                    hilight = at.createTransformedShape(hilight);
352                    g2.setColor(HIGHLIGHT_COLOR);
353                    g2.fill(hilight);
354                }
355                if (getForeground() != null) g2.setColor(getForeground());
356                tl.draw(g2, originX, originY);
357                cursorOff = new GreenfootImage(getImage());
358                if (isFocusOwner()) {
359                    // Draw the caret at the insertion index
360                    Shape[] carets = tl.getCaretShapes(index2);
361                    g2.setColor(CARET_COLOR);
362                    Shape caret = at.createTransformedShape(carets[0]);
363                    g2.draw(caret);
364                }
365                cursorOn = getImage();
366            } else if (isFocusOwner()) {
367                // Draw a caret for the empty box
368                cursorOff = new GreenfootImage(getImage());
369                g2.setColor(CARET_COLOR);
370                Insets i = getInsets();
371                g2.drawLine(i.left + X_PAD, i.top, i.right + X_PAD,
372                    getHeight() - i.bottom);
373                cursorOn = getImage();
374            }
375        }

377        /**
378         * Scrolls the field left or right until it is visible.
379         */
380        private void scrollToVisible() {
381            if (layout == null) return;
382            Insets i = getInsets();
383            int xMin = i.left + X_PAD;
384            int xMax = getWidth() - i.right - X_PAD;
385            int cursorX = layout.getCaretShapes(index2)[0].getBounds().x;
386            float adjustX = layout.getAdvance() - xMax + originX;
387            if (originX < xMin && adjustX < 0) {
388                // adjust right margin when characters deleted
389                originX = originX - adjustX;
390                if (originX > xMin) originX = xMin;
391            }
392            if (cursorX < xMin - originX) {
393                // Scroll to the left
394                originX = xMin - cursorX;
395            } else if (cursorX > xMax - originX) {
396                // Scroll to the right
397                originX = xMax - cursorX;
398            }
399        }

401        /**
402         * Returns a text layout for the text that the user has entered. The text
403         * layout is cached until <code>invalidateTextLayout</code> is called.
404         *
405         * @param g  the <code>Graphics</code> context in which to paint.
406         * @return a <code>TextLayout</code> for the text that the user has
407         *         entered, or <code>null</code> if no text has been entered.
408         */
409        private TextLayout getTextLayout(Graphics g) {
410            if (!validTextLayout) {
411                layout = null;
412                if (input.length() > 0) {
413                    Graphics2D g2 = (Graphics2D) g;
414                    FontRenderContext frc = g2.getFontRenderContext();
415                    layout = new TextLayout(getText(), getFont(), frc);
416                    originY = (getHeight() + layout.getAscent()
417                               - layout.getDescent()) / 2;
418                }
419            }
420            validTextLayout = true;
421            return layout;
422        }

424        /**
425         * Computes the character position of the mouse cursor.
426         *
427         * @param mouseX The x-coordinate of the mouse pointer.
428         * @param mouseY The y-coordinate of the mouse pointer.
429         * @return The character position of the mouse cursor.
430         */
431        private int getHitPosition(int mouseX, int mouseY) {
432            // Compute mouse click location relative to text layout's origin
433            float x = mouseX - getX() + getWidth() / 2 - originX;
434            float y = mouseY - getY() + getHeight() / 2;
435            // Get the character position of the mouse click.
436            TextHitInfo hit = layout.hitTestChar(x, y);
437            return hit.getInsertionIndex();
438        }
439    
440        /**
441         * Deletes the characters within the highlighted area.
442         */
443        private void deleteHighlighedChars() {
444            if (index2 > index1) {
445                input.delete(index1, index2);
446                index2 = index1;
447            } else if (index1 > index2) {
448                input.delete(index2, index1);
449                index1 = index2;
450            }
451        }
